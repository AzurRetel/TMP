name: 🚨 Weblogic Alert - Local Demo Workflow
description: Captures Weblogic ML anomaly alerts and indexes them locally to bypass firewall restrictions.
enabled: true
tags: ["demo", "weblogic", "anomaly-detection"]
consts:
  indexName: weblogic-alerts-demo
triggers:
  - type: alert
steps:
  - name: check_if_index_exists
    type: elasticsearch.indices.exists
    with:
      index: "{{ consts.indexName }}"

  - name: create_index_if_missing
    type: if
    condition: 'steps.check_if_index_exists.output == false'
    steps:
      - name: no_index_found_log
        type: console
        with:
          message: "Index {{ consts.indexName }} not found. Creating it now."
      - name: create_demo_index
        type: elasticsearch.indices.create
        with:
          index: "{{ consts.indexName }}"
          mappings:
            properties:
              rule_name: { type: keyword }
              message: { type: text }
              alert_id: { type: keyword }

  - name: log_incoming_alert
    type: console
    with:
      message: 'Received {{ event.alerts | size }} alert(s) from rule "{{ event.rule.name }}". Proceeding to index.'

  - name: index_alert_data
    type: elasticsearch.request
    with:
      method: POST
      path: /{{ consts.indexName }}/_doc
      headers:
        Content-Type: application/json
      body: |
        {
          "rule_name": "{{ event.rule.name }}",
          "alert_id": "{{ event.alerts[0]._id }}",
          "space_id": "{{ event.spaceId }}",
          "message": "Demo: Se detectó un comportamiento anómalo en Weblogic."
        }
